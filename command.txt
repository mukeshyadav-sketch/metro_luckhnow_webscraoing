    s1.st_name AS from_station,
    s2.st_name AS to_station,
    f.fare
FROM fares f
JOIN stations s1 ON f.from_station_id = s1.id
JOIN stations s2 ON f.to_station_id = s2.id
WHERE s1.st_name = 'CCS AIRPORT'
